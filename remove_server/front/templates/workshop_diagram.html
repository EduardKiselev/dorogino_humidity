function updateDiagramWithData(sensors) { // Get all sensor IDs that have data const sensorIdsWithData = new Set(sensors.map(sensor => sensor.sensor_id)); // Get all sensor markers currently on the diagram const allMarkers = document.querySelectorAll('.sensor-marker'); // Update existing markers - keep those without data but mark as "No data" allMarkers.forEach(marker => { const sensorId = parseInt(marker.dataset.sensorId); if (!sensorIdsWithData.has(sensorId)) { // Sensor has no data at this time, update to show "No data" const card = marker.querySelector('.sensor-info-card'); if (card) { const basicInfo = `Sensor ${sensorId}
No data`; card.innerHTML = basicInfo; } } }); // Update or add markers for sensors with data sensors.forEach(sensor => { let marker = document.querySelector(`.sensor-marker[data-sensor-id="${sensor.sensor_id}"]`); if (!marker) { // Create new marker if it doesn't exist marker = document.createElement('div'); marker.className = 'sensor-marker'; marker.dataset.sensorId = sensor.sensor_id; marker.style.cssText = `position: absolute; left: ${sensor.x}%; top: ${sensor.y}%; transform: translate(-50%, -50%); cursor: pointer;`; marker.title = sensor.description; document.getElementById('diagram-container').appendChild(marker); } // Update the content of the marker with new data marker.innerHTML = `
Sensor ${sensor.sensor_id}
T: ${sensor.temperature ? sensor.temperature.toFixed(1) + '°C' : 'No data'}
H: ${sensor.humidity ? sensor.humidity.toFixed(1) + '%' : 'No data'}
`; }); }
